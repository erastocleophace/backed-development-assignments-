const express = require('express');
const app = express();

app.use(express.json());

app.get('/', (req, res) => {
    res.send('Backend Server is Running');
});

app.get('/api/users', (req, res) => {
    res.json([
        { id: 1, name: "Erasto" },
        { id: 2, name: "Cleophace" }
    ]);
});

const PORT = 3000;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
